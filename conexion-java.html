            package Connections;
            import java.sql.*;
            import java.util.logging.Level;
            import java.util.logging.Logger;

            /**
            *
            * @author RICHARD
            */
            public class ConexionDB {
                public static Connection getConneccion(){
                try {
                        Class.forName("com.mysql.jdbc.Driver");  //DRIVER DE SQL DEVELOPER
                        String myDB = "jdbc:mysql://localhost:3306/colegio";//URL DE SQL DEVELOPER
                        String usuario="root";
                        String password="";
                        Connection cnx = DriverManager.getConnection(myDB,usuario,password);
                        return cnx;
                        
                    } catch (SQLException ex) {
                        System.out.println(ex.getMessage());
                    } catch (ClassNotFoundException ex) {
                        Logger.getLogger(ConexionDB.class.getName()).log(Level.SEVERE, null, ex);
                    }
                    return null;
                }  
            }

                    package modelo;
                    /**
                    *
                    * @author RICHARD
                    */
                    public class  Distrito {
                        String coddis,nomdis;

                        public Distrito() {
                        }// fin del contructor sin parametro
                    public Distrito(String coddis, String nomdis) {
                            this.coddis = coddis;
                            this.nomdis = nomdis;
                        }// fin del contructor con parametro
                        //los metodos get & set
                        public String getCoddis() {
                            return coddis;
                        }
                    public void setCoddis(String coddis) {
                            this.coddis = coddis;
                        }
                    public String getNomdis() {
                            return nomdis;
                        }
                    public void setNomdis(String nomdis) {
                            this.nomdis = nomdis;
                        }
                        }//fin de la clase Distrito

                    package Operacion;

                    import Connections.ConexionDB; //importamos la clase ConexionDB
                    import java.sql.*; //importamos la libreria SQL todo *
                    import java.util.ArrayList; //importar ArrayList
                    import modelo.Distrito; importamos la clase Distrito
                    
                    /**
                    *
                    * @author RICHARD
                    */
                    public class DistritoDB {
                    /*variables glovales de diferentes tipos*/
                        Connection cnx=null;       //variable de tipo de Connection
                        Statement st = null;  //variable de tipo de Statement
                        ResultSet rs = null;  //variable de tipo de ResultSet
                        PreparedStatement pst =null; //variable de tipo de PreparStatement
                    
                    //metodo para listar 
                    public ArrayList< Distrito> getListar(){
                         ArrayList< Distrito> distrito=new  ArrayList<>();//objeto de tipo array
                        try {
                            cnx=ConexionDB.getConneccion();
                            st=cnx.createStatement();
                            rs=st.executeQuery("select coddis,nomdis from distritos "); //Consulta a la DB

                            while(rs.next()){
                            Distrito dis=new Distrito();//objeto de tipo Distrito
                            dis.setCoddis(rs.getString(1));
                            dis.setNomdis(rs.getString(2));
                            distrito.add(dis);
                            }
                        } catch (SQLException e) {
                            System.out.println(e.getMessage()+"error listar");
                        }return distrito;
                    }
                    }
