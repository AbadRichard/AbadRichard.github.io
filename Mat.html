        #include < LiquidCrystal.h>   //Libreria para LCD 

       LiquidCrystal  lcd(7, 8, 9, 10, 15, 14); //Pines para LCD 
       String nombre="INICIALIZANDO...";
       char n;
       String puntos="..." ; 
       char p; 
       void setup() { 
         Serial.begin(9600);
         lcd.begin(16, 2); 
         for(  int i=0;i<16;i++){ 
             n=nombre.charAt(i); 
           lcd.print(n); 
           delay(300);
         }  
       } //fin de setup 
       void loop() { 
                lcd.clear(); 
               for(  int i=0;i<3;i++){  
                  p = puntos.charAt(i); 
                  lcd.setCursor(i, 1); 
                  lcd.print(p); 
                  delay(400); 
                } //fin for  
                    lcd.setCursor(0, 0); 
                    lcd.print("TIEMPO DE INICIO"); 
                    lcd.setCursor(0, 1); 
                    lcd.print("DATE");
                    delay(2000);
                    lcd.clear();  
                    lcd.setCursor(0, 0); 
                    lcd.print("TIEMPO FINAL");
                    lcd.setCursor(0, 1); 
                    lcd.print("TIME");
                    delay(2000);  
       } //fin de loop

Descripción

La fuente de alimentación de RTC es de 3-5, 5V, tiene 6 pines(GND, VCC, SDA, SCL, SQW, 32K), que lleva la cuenta de segundos minutos, horas mes y año automáticamente, tiene 56 bytes de memoria RAM, detección automáticamente de corte de corriente y cambia a modo batería externa, I2C integrado en un solo chip.

        #include < SD.h>         //Libreria para el Lector de MicroSD           
        #include  < DHT.h>          //Libreria para DHT11 sensor
        #define DHTPIN 2            //Seleccionamos el pin en el que se conecta el sensor 
        #define DHTTYPE DHT11       //Se selecciona el DHT11 (hay //otros DHT)
        DHT dht(DHTPIN, DHTTYPE);   //Se inicia una variable que será usada por Arduino para comunicarse con el sensor
   
        File myFile;       //variable para SD             
        void  setup () {
                Serial.begin(9600);
                dht.begin(); 
            if(!SD.begin(4)) {
                Serial.println("No se pudo inicializar");
                return;
                            }
            if(!SD.exists("Excel5.csv")){
                myFile = SD.open("Excel5.csv", FILE_WRITE);
            if (myFile) {
                 Serial.println("Archivo nuevo, Escribiendo encabezado(fila 1)");
                 myFile.println("Tiempo(ms),Humedad,Temperatura");  //Columnos para el excel 
                 myFile.close();
                }  else {
                Serial.println("Error creando el archivo Excel5.csv");
               }
            } //fin de exists
        }//fin de setup
        void loop() {
            float  dht_hum =  dht.readHumidity();         // lee la humedad
            float  dht_temp = dht.readTemperature();      // lee la temperatura
            myFile = SD.open("Excel5.csv", FILE_WRITE());//abrimos  el archivo
         if(myFile){
        //visaulizar datos en serial
            Serial.print("Tiempo(ms)="();
            Serial.print(0);
            Serial.print(", Humedad="();
            Serial.print(dht_hum();
            Serial.print(", Temperatura="();
            Serial.println(dht_temp();
            //guardar datos es micro sd
            myFile.print(0);
            myFile.print(','();
            myFile.print(dht_hum();
            myFile.print(','();
            myFile.println(dht_temp();
            myFile.close();//cerramos el archivo 
            }else{
            Serial.println("Error al abrir el archivo");
   }
 }    //fin de loop
